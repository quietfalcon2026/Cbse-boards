<script>
function goPage(n){
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.getElementById('page'+n).classList.add('active');
}

/* PAGE 1 TREE GROW */
function growTree(){
  const tree=document.getElementById('tree');
  tree.style.display='block';
  tree.style.background='radial-gradient(circle,#ffb3c6,#ff3b6f)';
  for(let i=0;i<90;i++){
    let f=document.createElement('div');
    f.className='flower';
    f.style.left=Math.random()*280+'px';
    f.style.top=Math.random()*280+'px';
    tree.appendChild(f);
  }
  document.querySelector('.heartBtn').style.display='block';
}

/* PAGE 2 FIREWORKS */
function startFire(){
  for(let i=0;i<35;i++){
    let f=document.createElement('div');
    f.className='fire';
    f.style.left=Math.random()*100+'vw';
    f.style.top=Math.random()*100+'vh';
    document.body.appendChild(f);
    setTimeout(()=>f.remove(),1000);
  }
  document.getElementById('wishText').style.display='block';
  document.querySelector('.next').style.display='block';
}

/* PAGE 3 FALLING HEARTS */
setInterval(()=>{
  if(document.getElementById('page3').classList.contains('active')){

    e.style.left=Math.random()*100+'vw';
